# 04. Convert Radar PPI to WRFDA ob.radar

Overview

This module converts raw Furuno ODIM HDF5 radar volume scans into the observation format required by WRFDA Radar Data Assimilation.
https://docs.wradlib.org/projects/notebooks/en/latest/notebooks/fileio/backends/odim_backend.html

Objectives

The objective of this module is to

- read Furuno ODIM HDF5 radar volume scans,
- extract radar reflectivity (DBZH) and radial velocity (VRAD),
- interpolate radar observations onto a regular Cartesian grid,
- convert the gridded observations into the WRFDA FM-128 Radar Observation format,
- automatically generate domain-specific observation files,
- optionally upload the observation files to an FTP server for operational forecasting.

The conversion consists of two sequential processing steps:

## Step 4. Convert Radar PPI to NetCDF

The first script reads the Furuno ODIM HDF5 files and converts each radar volume scan into two NetCDF files:

Reflectivity (dBZ)
Radial Velocity (VRAD)

Key processing steps include:
Reading ODIM HDF5 using xradar
Georeferencing radar coordinates using wradlib
Converting polar coordinates to a regular Cartesian grid
Applying a reflectivity threshold (20 dBZ)
Saving the output as compressed NetCDF files

script ppi2nc.py
```python
# -*- coding: utf-8 -*-
"""
Read Furuno ODIM HDF5
Inspect dataset and create first PPI

@author: Abdul Hamid Al H
"""

import os
import glob
import re

import numpy as np
import datetime as dt

import xradar as xd
import wradlib as wrl

from netCDF4 import Dataset


# =====================================================
# CONFIG
# =====================================================

indir = r"E:\WRFRUC\sample_radarbibhf5\2056_20260608"
outdir = r"E:\WRFRUC\sample_radarbibhf5\2056_20260608"
#indir = "/home/metpublic/task_release/HAMID/radar_bib/unduh_hdf5bib"
#outdir = "/home/metpublic/task_release/HAMID/radar_bib/ppi_2_ascii/ppi_2_nc"

os.makedirs(outdir, exist_ok=True)

grid_res = 750.0
max_range = 70000.0

R = 6371000.0


# =====================================================
# WRITE NC
# =====================================================

def write_nc(
    outfile,
    varname,
    data3d,
    lons,
    lats,
    levs,
    srcfile
):

    nc = Dataset(
        outfile,
        "w",
        format="NETCDF4"
    )

    nc.Conventions = "CF-1.6"

    nc.createDimension("lon", len(lons))
    nc.createDimension("lat", len(lats))
    nc.createDimension("lev", len(levs))

    lono = nc.createVariable(
        "lon",
        "f4",
        ("lon",)
    )

    lato = nc.createVariable(
        "lat",
        "f4",
        ("lat",)
    )

    levo = nc.createVariable(
        "lev",
        "f4",
        ("lev",)
    )

    lono[:] = lons
    lato[:] = lats
    levo[:] = levs

    lono.units = "degrees_east"
    lato.units = "degrees_north"
    levo.units = "degree"

    vo = nc.createVariable(
        varname,
        "f4",
        ("lat","lon","lev"),
        zlib=True,
        complevel=4,
        fill_value=np.nan
    )

    if varname == "dBZ":

        vo.long_name = "Reflectivity"
        vo.units = "dBZ"

    else:

        vo.long_name = "Radial Velocity"
        vo.units = "m s-1"

    vo[:] = data3d

    nc.comment = srcfile

    nc.close()


# =====================================================
# FILE LOOP
# =====================================================

files = sorted(
    glob.glob(
        os.path.join(indir, "*.h5")
    )
)

for fpath in files:

    fname = os.path.basename(fpath)

    m = re.search(
        r'(\d{8})_(\d{6})',
        fname
    )

    if m is None:
        continue

    datestr = m.group(1)
    timestr = m.group(2)

    minute = int(timestr[2:4])

    # ==========================================
    # HANYA KELIPATAN 10 MENIT
    # ==========================================

    if minute % 10 != 0:
        continue

    
    timestep = f"{datestr}_{timestr}"

    out_dbz = os.path.join(
        outdir,
        f"{timestep}_dbz.nc"
    )
    
    out_v = os.path.join(
        outdir,
        f"{timestep}_v.nc"
    )
    
    
    # ==========================================
    # SKIP JIKA SUDAH PERNAH DICONVERT
    # ==========================================
    if (
        os.path.exists(out_dbz)
        and os.path.getsize(out_dbz) > 0
        and os.path.exists(out_v)
        and os.path.getsize(out_v) > 0
    ):
    
        print("")
        print("================================")
        print("SKIP :", fname)
        print("DBZ dan VRAD sudah ada")
        print("================================")
    
        continue
    
    print("")
    print("================================")
    print("PROCESS :", fname)
    print("================================")
    
    t0 = dt.datetime.now()
    
    # =====================================================
    # OPEN RADAR
    # =====================================================
    
    try:
    
        vol = xd.io.open_odim_datatree(
            fpath
        )
    
    except Exception as e:
    
        print("")
        print("================================")
        print(f"ERROR OPENING : {fname}")
        print(e)
        print("================================")
    
        continue
    
    radar_lat = float(
        vol.latitude.values
    )
    
    radar_lon = float(
        vol.longitude.values
    )
    
    levs = vol["sweep_fixed_angle"].values

    # =====================================================
    # GRID
    # =====================================================

    xgrid = np.arange(
        -max_range,
        max_range + grid_res,
        grid_res
    )

    ygrid = np.arange(
        -max_range,
        max_range + grid_res,
        grid_res
    )

    xx, yy = np.meshgrid(
        xgrid,
        ygrid
    )

    lonGrid = radar_lon + (
        xx /
        (
            R *
            np.cos(
                np.deg2rad(radar_lat)
            )
        )
    ) * 180.0 / np.pi

    latGrid = radar_lat + (
        yy / R
    ) * 180.0 / np.pi

    lons = lonGrid[0,:].astype(np.float32)
    lats = latGrid[:,0].astype(np.float32)

    nlat = len(lats)
    nlon = len(lons)
    nlev = len(levs)

    dbz3d = np.full(
        (nlat,nlon,nlev),
        np.nan,
        dtype=np.float32
    )

    vrad3d = np.full(
        (nlat,nlon,nlev),
        np.nan,
        dtype=np.float32
    )

    # =====================================================
    # SWEEP LOOP
    # =====================================================

    for sw in range(nlev):

        print(
            f"sweep={sw} "
            f"elev={levs[sw]}"
        )

        ds = vol[
            f"sweep_{sw}"
        ].to_dataset(
            inherit="all_coords"
        )

        ds = ds.wrl.georef.georeference()

        x = ds.x.values
        y = ds.y.values

        dbz = ds.DBZH.values.astype(
            np.float32
        )

        vrad = ds.VRAD.values.astype(
            np.float32
        )

        # -----------------------------------
        # DONALDI THRESHOLD
        # -----------------------------------

        dbz[dbz < 20.0] = np.nan

        # -----------------------------------
        # GRID INDEX
        # -----------------------------------

        ix = np.round(
            (x - xgrid[0]) /
            grid_res
        ).astype(int)

        iy = np.round(
            (y - ygrid[0]) /
            grid_res
        ).astype(int)

        valid = (
            (ix >= 0)
            &
            (ix < len(xgrid))
            &
            (iy >= 0)
            &
            (iy < len(ygrid))
        )

        ix = ix[valid]
        iy = iy[valid]

        dbzv = dbz[valid]
        vradv = vrad[valid]

        grd_dbz = np.full(
            xx.shape,
            np.nan,
            dtype=np.float32
        )

        grd_v = np.full(
            xx.shape,
            np.nan,
            dtype=np.float32
        )

        for k in range(len(ix)):

            # DBZH

            if np.isfinite(dbzv[k]):

                if np.isnan(
                    grd_dbz[iy[k],ix[k]]
                ):

                    grd_dbz[
                        iy[k],
                        ix[k]
                    ] = dbzv[k]

                else:

                    grd_dbz[
                        iy[k],
                        ix[k]
                    ] = max(
                        grd_dbz[
                            iy[k],
                            ix[k]
                        ],
                        dbzv[k]
                    )

            # VRAD

            grd_v[
                iy[k],
                ix[k]
            ] = vradv[k]

        dbz3d[:,:,sw] = grd_dbz
        vrad3d[:,:,sw] = grd_v

    # =====================================================
    # OUTPUT
    # =====================================================

    out_dbz = os.path.join(
        outdir,
        f"{timestep}_dbz.nc"
    )
    
    out_v = os.path.join(
        outdir,
        f"{timestep}_v.nc"
    )

    write_nc(
        out_dbz,
        "dBZ",
        dbz3d,
        lons,
        lats,
        levs,
        fname
    )

    write_nc(
        out_v,
        "VRAD",
        vrad3d,
        lons,
        lats,
        levs,
        fname
    )

    print("DBZ :", out_dbz)
    print("VRAD:", out_v)
    print(
        "Elapsed:",
        dt.datetime.now() - t0
    )

print("")
print("ALL FINISHED")
```

## Step 5. Convert NetCDF to WRFDA ob.radar
The second script converts the NetCDF radar files into the observation format required by WRFDA.
Please check this for Example of ob.radar data format:
- https://www2.mmm.ucar.edu/wrf/users/wrfda/Tutorials/2015_Aug/docs/WRFDA_Radar.pdf

Major processing steps include:

Reading DBZH and VRAD NetCDF files
Matching observations to each WRF domain
Computing observation heights using radar beam geometry
Generating FM-128 radar observation records
Writing the final ob.radar files
Optionally uploading the files to an FTP server

nc2obradar.py script:
```python
from ftplib import FTP
import os
import numpy as np
import netCDF4 as nc
from datetime import datetime, timedelta

# =========================
# CONFIG
# =========================
# ctl_dir = r"E:\WRFRUC\bib_dat"
# nc_dir = r"E:\WRFRUC\sample_radarbibhf5\2056_20260608"
# os.chdir(ctl_dir)

#ctl_dir = "/home/metpublic/task_release/HAMID/radar_bib/ppi_2_ascii/ctl_domain_wrfruc"

#nc_dir = "/home/metpublic/task_release/HAMID/radar_bib/ppi_2_ascii/ppi_2_nc"

#out_dir = "/home/metpublic/task_release/HAMID/radar_bib/ppi_2_ascii/nc_2_ascii"

ctl_dir = r"E:\WRFRUC\bib_dat"
nc_dir = r"E:\WRFRUC\sample_radarbibhf5\2056_20260608"

out_dir = r"E:\WRFRUC\sample_radarbibhf5\2056_20260608"

os.makedirs(out_dir, exist_ok=True)
done_dir = os.path.join(out_dir, ".done")
os.makedirs(done_dir, exist_ok=True)


radar_name = "BIB"
radar_lon = 115.6579
radar_lat = -3.5966
radar_alt = 37

# FTP TARGET
FTP_HOST = "-"
FTP_USER = "-"
FTP_PASS = "-"
FTP_DIR  = "-"


# =========================
# FILE LIST
# =========================
nczfilelist = sorted([os.path.join(nc_dir, f) for f in os.listdir(nc_dir) if f.endswith("_dbz.nc")])
ncvfilelist = sorted([os.path.join(nc_dir, f) for f in os.listdir(nc_dir) if f.endswith("_v.nc")])
ctlfilelist = sorted([
    os.path.join(ctl_dir, f)
    for f in os.listdir(ctl_dir)
    if f.endswith(".ctl")
])

# =========================
# TIME BINNING ±20 MIN
# =========================
def get_center_hour(date_str, time_str):
    raw = datetime.strptime(date_str + time_str[:4], "%Y%m%d%H%M")

    lower = raw.replace(minute=0, second=0, microsecond=0)
    upper = lower + timedelta(hours=1)

    diff_lower = abs((raw - lower).total_seconds()) / 60
    diff_upper = abs((raw - upper).total_seconds()) / 60

    if diff_lower <= diff_upper:
        chosen = lower
        diff = diff_lower
    else:
        chosen = upper
        diff = diff_upper

    if diff > 20:
        return None

    return chosen

# =========================
# READ CTL GRID
# =========================
def read_ctl_coordinates(filename):
    with open(filename, "r") as f:
        lines = f.readlines()

    data = [float(x.strip()) for x in lines if x.startswith(" ")]
    half = len(data) // 2

    lonc = np.array(data[:half])
    latc = np.array(data[half:])

    return lonc, latc

# =========================
# FTP UPLOAD
# =========================

def ftp_upload(local_file):

    try:

        ftp = FTP(FTP_HOST, timeout=30)
        ftp.login(FTP_USER, FTP_PASS)
        ftp.cwd(FTP_DIR)

        remote_name = os.path.basename(local_file)

        try:
            ftp.delete(remote_name)
        except:
            pass

        with open(local_file, "rb") as fp:
            ftp.storbinary(
                f"STOR {remote_name}",
                fp
            )

        ftp.quit()

        print(f"FTP OK : {remote_name}")

        return True

    except Exception as e:

        print(f"FTP FAIL : {local_file}")
        print(e)

        return False
        
# =========================
# MAIN LOOP
for ncfname in nczfilelist:

    nc1fname = ncfname.replace(
        "_dbz.nc",
        "_v.nc"
    )

    if not os.path.exists(nc1fname):
        print(f"MISSING VRAD : {nc1fname}")
        continue

    for ctlfile in ctlfilelist:

        lonc, latc = read_ctl_coordinates(ctlfile)


        basename = os.path.basename(ncfname).replace(".nc", "")
        parts = basename.split("_")

        date_str = parts[0]   # 20260608
        time_str = parts[1]   # 065000
        
        timestep = f"{date_str}_{time_str}"
        

        domain_name = os.path.splitext(os.path.basename(ctlfile))[0]
        
        done_file = os.path.join(
            done_dir,
            f"{timestep}_{domain_name}.done"
        )
        
        if os.path.exists(done_file):
        
            print(
                f"SKIP : {domain_name} {timestep}"
            )
        
            continue
        
        
        print("")
        print("==================================================")
        print(f"DOMAIN : {domain_name}")
        print(f"DBZ    : {os.path.basename(ncfname)}")
        print(f"VRAD   : {os.path.basename(nc1fname)}")
        print("==================================================")
        
        if os.path.exists(done_file):
        
            print("")
            print("================================")
            print(f"SKIP NC2ASCII : {timestep}")
            print("ALREADY PROCESSED")
            print("================================")
        
            continue

        # =========================
        # TIME BINNING FILTER
        # =========================
        center_time = get_center_hour(date_str, time_str)

        if center_time is None:
            print(f"SKIP OUT OF WINDOW: {basename}")
            continue

        file_time = center_time.strftime("%Y%m%d_%H%M")

        obs_time = datetime.strptime(
            date_str + time_str[:6],
            "%Y%m%d%H%M%S"
        ).strftime("%Y-%m-%d_%H:%M:%S")

        # =========================
        # OPEN NETCDF
        # =========================
        try:

            ds = nc.Dataset(ncfname)
            ds1 = nc.Dataset(nc1fname)
        
        except Exception as e:
        
            print("")
            print("================================")
            print(f"CORRUPT NC : {os.path.basename(ncfname)}")
            print(e)
            print("================================")
        
            continue

        lon = ds.variables["lon"][:]
        lat = ds.variables["lat"][:]
        lev = ds.variables["lev"][:]

        dbz_array = np.ma.filled(ds.variables["dBZ"][:], np.nan)
        rv_array  = np.ma.filled(ds1.variables["VRAD"][:], np.nan)

        ds.close()
        ds1.close()

        # =========================
        # DOMAIN FILTER
        # =========================
        idlonstart, idlonend = 0, len(lonc)
        idlatstart, idlatend = 0, len(latc)

        if lonc[0] < lon[0]:
            idlonstart = np.where(lonc > lon[0])[0][0]
        if lonc[-1] > lon[-1]:
            idlonend = np.where(lonc < lon[-1])[0][-1]
        if latc[0] < lat[0]:
            idlatstart = np.where(latc > lat[0])[0][0]
        if latc[-1] > lat[-1]:
            idlatend = np.where(latc < lat[-1])[0][-1]

        linedata = []
        alt = []

        # =========================
        # GRID LOOP
        # =========================
        for ilon in lonc[idlonstart:idlonend]:
            for jlat in latc[idlatstart:idlatend]:

                close_lon = np.argmin(np.abs(lon - ilon))
                close_lat = np.argmin(np.abs(lat - jlat))

                valid_levels = np.where(~np.isnan(dbz_array[close_lon, close_lat, 1:]))[0]
                nalt = len(valid_levels)

                if nalt > 0:
                    alt.append(nalt)

                    linedata.append(
                        f"FM-128 RADAR   "
                        f"{obs_time:<19}"
                        f"{jlat:13.3f}"
                        f"{ilon:14.3f}"
                        f"{radar_alt:10.1f}"
                        f"{nalt:8d}"
                    )

                    for kalt in valid_levels[::-1]:
                        k = kalt + 1

                        rf_data = float(dbz_array[close_lon, close_lat, k])
                        rv_data = float(rv_array[close_lon, close_lat, k])

                        if np.isnan(rf_data):
                            rf_data, rf_qc, rf_err = -888888.0, -88, 1.0
                        else:
                            rf_qc, rf_err = 0, 1.0

                        if np.isnan(rv_data):
                            rv_data, rv_qc, rv_err = -888888.0, -88, 1.0
                        else:
                            rv_qc, rv_err = 0, 1.0

                        delta_lat = radar_lat - jlat
                        delta_lon = radar_lon - ilon

                        degtometer = 111320
                        Rtopography = np.sqrt(delta_lon**2 + delta_lat**2) * degtometer / 1000.0

                        re = 6370.0
                        scan_angle = lev[k]

                        R = Rtopography / np.cos(np.deg2rad(scan_angle))
                        H = (R * np.sin(np.deg2rad(scan_angle)) + (R**2) / (2 * re)) * 1000.0

                        linedata.append(
                            f"   {H:12.2f}{rv_data:12.3f}{rv_qc:4d}{rv_err:12.3f}  "
                            f"{rf_data:12.3f}{rf_qc:4d}{rf_err:12.3f}"
                        )

        # =========================
        # OUTPUT NAME
        # =========================

        outfname = os.path.join(
            out_dir,
            f"{file_time}_{domain_name}_obradar"
        )
        
        max_alt = max(alt) if len(alt) > 0 else 0
        with open(outfname, "w") as f:
            f.write(f"TOTAL NUMBER =  {1}\n")
            f.write("#-----------------#\n\n")

            f.write(
                f"RADAR  {radar_name:^12}"
                f"{radar_lon:10.3f}"
                f"{radar_lat:10.3f}"
                f"{radar_alt:10.1f}  "
                f"{obs_time:<19}"
                f"{len(alt):6d}"
                f"{max_alt:6d}\n"
            )

            f.write("#-------------------------------------------------------------------------------#\n\n")
            f.write("\n".join(linedata))

        print(f"Saved {outfname}")
        
        if ftp_upload(outfname):
        
            with open(done_file, "w") as fp:
                fp.write(datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S"))
        
            print(f"DONE MARKER : {done_file}")
```
