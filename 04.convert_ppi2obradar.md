# Module 04 Convert Raw Radar Data into ob.radar for WRFDA Input

## 04a. Convert Radar PPI-VAISALA Uganda to WRFDA `ob.radar`

## Overview

This module explains how to convert **Vaisala IRIS RAW radar data** into the **WRFDA radar observation (`ob.radar`)** format.

The conversion process is divided into two Python scripts:

1. **`vaisala_ppi2nc.py`**
   - Reads Vaisala IRIS RAW files.
   - Extracts radar metadata.
   - Converts polar radar data into Cartesian grids.
   - Saves reflectivity (DBZ) and radial velocity (VRAD) into NetCDF files.

2. **`vaisala_nc2obradar.py`**
   - Reads the generated NetCDF files.
   - Matches radar observations with the WRF model grid.
   - Converts radar profiles into the FM-128 `ob.radar` format required by WRFDA.

The workflow is implemented using **wradlib**, an open-source Python library for weather radar processing.

Reference:

https://docs.wradlib.org/en/2.2.0/notebooks/fileio/legacy/read_iris.html

## Processing Workflow

The overall processing workflow is illustrated below.

```text
Vaisala IRIS RAW
        │
        ▼
vaisala_ppi2nc.py
        │
        ▼
Read radar metadata
        │
        ▼
Polar → Cartesian conversion
        │
        ▼
Create 3D NetCDF
(DBZ & VRAD)
        │
        ▼
vaisala_nc2obradar.py
        │
        ▼
Read geo_em.d0*
        │
        ▼
Nearest-grid matching (KDTree)
        │
        ▼
Generate FM-128 ob.radar
        │
        ▼
WRFDA Radar Assimilation
```

## Script 1. Convert RAW data [Vaisala] to NetCDF

Script vaisala_ppi2nc.py::

```python

#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Convert Vaisala IRIS RAW (*.RAW*)
to NetCDF

Part-1
Read RAW file
Read metadata
Prepare grid information
Author : Abdul Hamid Al Habib
"""

import os
import re
import glob
import datetime as dt

import numpy as np
import wradlib as wrl
from wradlib.ipol import Nearest
from scipy.spatial import cKDTree
from netCDF4 import Dataset

# =====================================================
# CONFIG
# =====================================================

indir = r"E:\Uganda\product_raw" #Edit the directory for raw data input
outdir = r"E:\Uganda\netcdf"  #edit the directory for netcdf output

os.makedirs(outdir, exist_ok=True)

# Cartesian Grid
grid_res = 750.0          # meter, you can edit based on your radar resolution
max_range = 75000.0       # meter, you can edit based on your radar max_range

R = 6371000.0             # Earth radius


# =====================================================
# WRITE NETCDF
# =====================================================

def write_nc(
    outfile,
    varname,
    data3d,
    lons,
    lats,
    levs,
    srcfile
):

    nc = Dataset(
        outfile,
        "w",
        format="NETCDF4"
    )

    nc.Conventions = "CF-1.6"

    nc.createDimension(
        "lon",
        len(lons)
    )

    nc.createDimension(
        "lat",
        len(lats)
    )

    nc.createDimension(
        "lev",
        len(levs)
    )

    lono = nc.createVariable(
        "lon",
        "f4",
        ("lon",)
    )

    lato = nc.createVariable(
        "lat",
        "f4",
        ("lat",)
    )

    levo = nc.createVariable(
        "lev",
        "f4",
        ("lev",)
    )

    lono[:] = lons
    lato[:] = lats
    levo[:] = levs

    lono.units = "degrees_east"
    lato.units = "degrees_north"
    levo.units = "degree"

    vo = nc.createVariable(
        varname,
        "f4",
        ("lat","lon","lev"),
        fill_value=np.nan,
        zlib=True,
        complevel=4
    )

    if varname == "dBZ":

        vo.long_name = "Reflectivity"
        vo.units = "dBZ"

    else:

        vo.long_name = "Radial Velocity"
        vo.units = "m s-1"

    vo[:] = data3d

    nc.comment = srcfile

    nc.close()


# =====================================================
# SEARCH RAW FILE
# =====================================================

files = sorted(

    glob.glob(
        os.path.join(
            indir,
            "*.RAW*"
        )
    )

)

print("")
print("=======================================")
print("Number of RAW files :", len(files))
print("=======================================")


# =====================================================
# LOOP FILE
# =====================================================

for fpath in files:

    fname = os.path.basename(fpath)

    print("")
    print("=======================================")
    print("PROCESS :", fname)
    print("=======================================")

    t0 = dt.datetime.now()

    # -------------------------------------------------
    # READ RAW
    # -------------------------------------------------

    try:

        iris = wrl.io.iris.read_iris(fpath)

    except Exception as e:

        print("FAILED :", e)
        continue


    # -------------------------------------------------
    # METADATA
    # -------------------------------------------------

    ph = iris["product_hdr"]["product_end"]

    radar_lat = ph["latitude"]

    radar_lon = ph["longitude"]

    radar_hgt = ph["radar_height_above_reference"]
    
    # -------------------------------------------------
    # NUMBER OF SWEEP
    # -------------------------------------------------
    
    nsweep = iris["nsweeps"]
    
    nrays = iris["nrays"]
    
    nbins = iris["nbins"]
    
    # -------------------------------------------------
    # ALL SWEEP PROCESS
    # -------------------------------------------------
    
    if nsweep != 7:
    
        print("")
        print("SKIP :", fname)
        print(f"Reason : only {nsweep} sweep (need 7 sweep)")
        print("")
    
        continue
    
    print("")
    
    print("Radar Latitude :", radar_lat)
    
    print("Radar Longitude:", radar_lon)
    
    print("Radar Height   :", radar_hgt)


    print("")

    print("Sweep :", nsweep)

    print("Ray   :", nrays)

    print("Gate  :", nbins)


    # -------------------------------------------------
    # RANGE INFO
    # -------------------------------------------------

    cfg = iris[
        "ingest_header"
    ][
        "task_configuration"
    ][
        "task_range_info"
    ]


    first_gate = (
        cfg["range_first_bin"] / 1000.0
    )

    gate_spacing = (
        cfg["step_output_bins"] / 100.0
    )

    print("")

    print("First Gate :", first_gate, "m")

    print("Gate Spacing :", gate_spacing, "m")


    ranges = (
        first_gate +
        np.arange(
            nbins
        ) * gate_spacing
    )


    # -------------------------------------------------
    # ELEVATION
    # -------------------------------------------------

    levs = []

    for sw in range(1, nsweep + 1):

        elev = np.nanmean(

            iris[
                "data"
            ][
                sw
            ][
                "sweep_data"
            ][
                "elevation"
            ]

        )

        levs.append(elev)

    levs = np.asarray(
        levs,
        dtype=np.float32
    )


    print("")

    print("Elevation")

    print(levs)


    # -------------------------------------------------
    # OUTPUT NAME
    # -------------------------------------------------

    tscan = ph["ingest_time"]

    timestep = tscan.strftime(
        "%Y%m%d_%H%M%S"
    )

    out_dbz = os.path.join(

        outdir,

        f"{timestep}_dbz.nc"

    )

    out_v = os.path.join(

        outdir,

        f"{timestep}_v.nc"

    )

    print("")

    print("Output DBZ :", out_dbz)

    print("Output VEL :", out_v)

    print("")

    # =====================================================
    # POLAR TO CARTESIAN GRID
    # =====================================================

    xgrid = np.arange(
        -max_range,
        max_range + grid_res,
        grid_res
    )
    
    ygrid = np.arange(
        -max_range,
        max_range + grid_res,
        grid_res
    )
    
    xx, yy = np.meshgrid(
        xgrid,
        ygrid
    )
    
    trg = np.column_stack(
    (
        xx.ravel(),
        yy.ravel()
    ))
    
    lonGrid = radar_lon + (
    
        xx /
    
        (
    
            R *
    
            np.cos(
                np.deg2rad(
                    radar_lat
                )
            )
    
        )
    
    ) * 180.0 / np.pi
    
    
    latGrid = radar_lat + (
    
        yy / R
    
    ) * 180.0 / np.pi
    
    
    lons = lonGrid[0,:].astype(
        np.float32
    )
    
    lats = latGrid[:,0].astype(
        np.float32
    )
    
    
    nlat = len(lats)
    
    nlon = len(lons)
    
    nlev = len(levs)
    
    
    dbz3d = np.full(
    
        (nlat,nlon,nlev),
    
        np.nan,
    
        dtype=np.float32
    
    )
    
    
    vrad3d = np.full(
    
        (nlat,nlon,nlev),
    
        np.nan,
    
        dtype=np.float32
    
    )
    
    
    # =====================================================
    # LOOP SWEEP
    # =====================================================
    
    for sw in range(nsweep):
    
        print("")
        print("------------------------------------")
        print(
            "Sweep",
            sw + 1,
            "/",
            nsweep,
            " Elev=",
            levs[sw]
        )
    
        sdata = iris[
            "data"
        ][
            sw + 1
        ][
            "sweep_data"
        ]
    
    
        # ---------------------------------------------
        # MOMENTS
        # ---------------------------------------------
        
        dbz = sdata["DB_DBZ"].astype(np.float32)
        
        vrad = np.ma.filled(
            sdata["DB_VEL"],
            np.nan
        ).astype(np.float32)
        
        dbz[dbz < 20] = np.nan
        
        # ---------------------------------------------
        # POLAR COORDINATE
        # ---------------------------------------------
        
        az = np.deg2rad(sdata["azimuth"])
        
        rr, aa = np.meshgrid(ranges, az)
        
        x = rr * np.sin(aa)
        y = rr * np.cos(aa)
        
        # ============================================
        # Reflectivity and Radial Velocity
        # ============================================
        
        src = np.column_stack(
            (
                x.ravel(),
                y.ravel()
            )
        )
        
        mask = np.isfinite(dbz.ravel())
        
        tree = cKDTree(
            src[mask]
        )
        
        dist, idx = tree.query(
            trg,
            distance_upper_bound=grid_res
        )
        
        grd_dbz = np.full(
            trg.shape[0],
            np.nan,
            dtype=np.float32
        )
        
        valid = np.isfinite(dist) & (idx < mask.sum())
        
        grd_dbz[valid] = dbz.ravel()[mask][idx[valid]]
        
        grd_dbz = grd_dbz.reshape(xx.shape)
        
        grd_dbz[grd_dbz < 0] = np.nan
        
        
        mask = np.isfinite(vrad.ravel())
        
        tree = cKDTree(
            src[mask]
        )
        
        dist, idx = tree.query(
            trg,
            distance_upper_bound=grid_res
        )
        
        grd_v = np.full(
            trg.shape[0],
            np.nan,
            dtype=np.float32
        )
        
        valid = np.isfinite(dist)
        
        grd_v[valid] = vrad.ravel()[mask][idx[valid]]
        
        grd_v = grd_v.reshape(xx.shape)
        # =====================================================
        # SAVE TO 3D ARRAY
        # =====================================================
        
        dbz3d[:,:,sw] = grd_dbz
        
        vrad3d[:,:,sw] = grd_v
    
    # =====================================================
    # WRITE OUTPUT
    # =====================================================

    write_nc(

        out_dbz,

        "dBZ",

        dbz3d,

        lons,

        lats,

        levs,

        fname

    )


    write_nc(

        out_v,

        "VRAD",

        vrad3d,

        lons,

        lats,

        levs,

        fname

    )


    print("")
    print("--------------------------------------")
    print("DBZ  :", out_dbz)
    print("VRAD :", out_v)
    print(
        "Elapsed :",
        dt.datetime.now() - t0
    )
    
print("")
print("====================================")
print("ALL FINISHED")
print("====================================")
```

## Script 2. Convert NetCDF [Vaisala] to WRFDA ob.radar
Please check this for Example of ob.radar data format:
- https://www2.mmm.ucar.edu/wrf/users/wrfda/Tutorials/2015_Aug/docs/WRFDA_Radar.pdf

Script vaisala_nc2obradar.py:

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
Convert Vaisala NetCDF -> WRFDA ob.radar
Part-2
Read geo_em
Read radar NetCDF
Author : Abdul Hamid Al Habib
"""

import os
from ftplib import FTP
from datetime import datetime,timedelta

import numpy as np
from netCDF4 import Dataset
from scipy.spatial import cKDTree

# =====================================================
# CONFIG
# =====================================================

geo_files=[
    r"E:\Uganda\geo_em.d01.nc", #Edit the directory for your geo_em.d01.nc input (from geogrid.exe step)
    r"E:\Uganda\geo_em.d02.nc", #Edit the directory for your geo_em.d02.nc input (from geogrid.exe step)
]

nc_dir=r"E:\Uganda\netcdf" #Edit the directory where netcdf file located (from scipt ppi2nc.py output)
out_dir=r"E:\Uganda\obradar" #Edit the directory for save the ob.radar output

os.makedirs(out_dir,exist_ok=True)

done_dir=os.path.join(out_dir,".done")
os.makedirs(done_dir,exist_ok=True)

radar_name="VSL"
radar_lon=32.43536001071334
radar_lat=0.03231005743145943
radar_alt=1127.

FTP_HOST="-"
FTP_USER="-"
FTP_PASS="-"
FTP_DIR="-"

# =====================================================
# FILE LIST
# =====================================================

dbz_files=sorted([
    os.path.join(nc_dir,f)
    for f in os.listdir(nc_dir)
    if f.endswith("_dbz.nc")
])

print("")
print("===================================")
print("Number of DBZ :",len(dbz_files))
print("===================================")

# =====================================================
# READ GEO_EM
# =====================================================

def read_geo_em(fname):

    nc=Dataset(fname)

    lon=nc.variables["XLONG_M"][0].astype(np.float32)
    lat=nc.variables["XLAT_M"][0].astype(np.float32)

    nc.close()

    return lon,lat

# =====================================================
# TIME BINNING
# =====================================================

def get_center_hour(date_str,time_str):

    raw=datetime.strptime(
        date_str+time_str[:4],
        "%Y%m%d%H%M"
    )

    lower=raw.replace(
        minute=0,
        second=0,
        microsecond=0
    )

    upper=lower+timedelta(hours=1)

    if abs(raw-lower)<=abs(raw-upper):
        t=lower
    else:
        t=upper

    if abs(raw-t).total_seconds()/60>20:
        return None

    return t

# =====================================================
# FTP
# =====================================================

def ftp_upload(localfile):

    try:

        ftp=FTP(FTP_HOST,timeout=30)

        ftp.login(
            FTP_USER,
            FTP_PASS
        )

        ftp.cwd(FTP_DIR)

        remote=os.path.basename(localfile)

        try:
            ftp.delete(remote)
        except:
            pass

        with open(localfile,"rb") as fp:
            ftp.storbinary(
                f"STOR {remote}",
                fp
            )

        ftp.quit()

        print("FTP OK :",remote)

        return True

    except Exception as e:

        print("FTP FAIL :",e)

        return False

# =====================================================
# MAIN LOOP
# =====================================================

for dbz_file in dbz_files:

    vel_file=dbz_file.replace(
        "_dbz.nc",
        "_v.nc"
    )

    if not os.path.exists(vel_file):

        print("Missing :",vel_file)

        continue

    base=os.path.basename(dbz_file).replace(".nc","")

    date_str,time_str=base.split("_")[:2]

    center_time=get_center_hour(
        date_str,
        time_str
    )

    if center_time is None:

        print("Skip :",base)

        continue

    
    obs_time = datetime.strptime(
        date_str + time_str,
        "%Y%m%d%H%M%S"
    ).strftime("%Y-%m-%d_%H:%M:%S")
    

    print("")
    print("===================================")
    print(base)
    print("===================================")

    # ---------------------------------
    # READ RADAR
    # ---------------------------------

    ds=Dataset(dbz_file)
    ds2=Dataset(vel_file)

    lon=ds.variables["lon"][:]
    lat=ds.variables["lat"][:]
    lev=ds.variables["lev"][:]

    dbz=np.ma.filled(
        ds.variables["dBZ"][:],
        np.nan
    )

    vrad=np.ma.filled(
        ds2.variables["VRAD"][:],
        np.nan
    )

    ds.close()
    ds2.close()
    # ---------------------------------
    # KDTree radar grid
    # ---------------------------------

    xx,yy=np.meshgrid(lon,lat)
    
    radar_points=np.column_stack((xx.ravel(),yy.ravel()))
    
    radar_tree=cKDTree(radar_points)
    
    radar_lat_idx,radar_lon_idx=np.indices(xx.shape)


    # ---------------------------------
    # LOOP DOMAIN
    # ---------------------------------

    for geo_file in geo_files:

        domain=os.path.basename(
            geo_file
        ).replace(
            "geo_em.",""
        ).replace(
            ".nc",""
        )


        glon,glat=read_geo_em(
            geo_file
        )


        wrf_points=np.column_stack(
            (
                glon.ravel(),
                glat.ravel()
            )
        )

        dist,idx=radar_tree.query(
            wrf_points
        )

        # ======================================================
        # PART-2
        # Convert KDTree index -> ob.radar profile
        # ======================================================
        
        out=[]
        alt=[]
        
        ny,nx=glon.shape
        
        for n in range(len(idx)):
        
            # -----------------------------
            # posisi grid WRF
            # -----------------------------
            wrf_i=n//nx
            wrf_j=n%nx
        
            lat0=float(glat[wrf_i,wrf_j])
            lon0=float(glon[wrf_i,wrf_j])
        
            # -----------------------------
            # posisi grid radar hasil KDTree
            # -----------------------------
            radar_index=idx[n]
        
            ilat=radar_index//len(lon)
            ilon=radar_index%len(lon)
        
            # -----------------------------
            # jarak radar-ke-grid
            # -----------------------------
            delta_lat=radar_lat-lat0
            delta_lon=radar_lon-lon0
            
            degtometer=111320.
            
            Rtopography=np.sqrt(
                delta_lon**2+
                delta_lat**2
            )*degtometer/1000.
        

            # -----------------------------
            # profile radar
            # -----------------------------
            profile=dbz[ilat,ilon,:]
        
            valid=np.where(
                ~np.isnan(
                    dbz[ilat,ilon,1:]
                )
            )[0]
        
            if len(valid)==0:
                continue
        
            nlev=len(valid)
            alt.append(nlev)
            
        
            # -----------------------------
            # HEADER FM128
            # -----------------------------
            
            
            out.append(
                f"FM-128 RADAR   "
                f"{obs_time:<19}"
                f"{lat0:13.3f}"
                f"{lon0:14.3f}"
                f"{radar_alt:10.1f}"
                f"{nlev:8d}"
            )

            # -----------------------------
            # LEVEL
            # -----------------------------
            #for k in valid[::-1]:
            for kk in valid[::-1]:

                k=kk+1
        
                rf=float(dbz[ilat,ilon,k])
                rv=float(vrad[ilat,ilon,k])
        
                rf_qc=0 if np.isfinite(rf) else -88
                rv_qc=0 if np.isfinite(rv) else -88
        
                if not np.isfinite(rf):
                    rf=-888888.
        
                if not np.isfinite(rv):
                    rv=-888888.
        
                elev=float(lev[k])
        
                re=6370.
        
                
                R=Rtopography/np.cos(np.deg2rad(elev))

                H=(R*np.sin(np.deg2rad(elev))
                 +(R*R)/(2*re))*1000.
        
                
                out.append(
                    f"   {H:12.2f}"
                    f"{rv:12.3f}"
                    f"{rv_qc:4d}"
                    f"{1.0:12.3f}  "
                    f"{rf:12.3f}"
                    f"{rf_qc:4d}"
                    f"{1.0:12.3f}"
                )
                
                
        
        outfile=os.path.join(
            out_dir,
            f"{center_time.strftime('%Y%m%d_%H%M')}_{domain}_obradar"
        )
        

        max_alt = max(alt) if len(alt) > 0 else 0
        with open(outfile,"w") as f:

            f.write(f"TOTAL NUMBER = {1}\n")
            f.write("#-----------------#\n\n")
            
            f.write(
            f"RADAR  {radar_name:^12}"
            f"{radar_lon:10.3f}"
            f"{radar_lat:10.3f}"
            f"{radar_alt:10.1f}  "
            f"{obs_time:<19}"
            f"{len(alt):6d}"
            f"{max_alt:6d}\n"
            )
            

            f.write("#-------------------------------------------------------------------------------#\n\n")

            f.write("\n".join(out))

        print("Saved :",outfile)
```

## 04b. Convert Radar PPI-Furuno to WRFDA ob.radar

Overview

This module converts raw Furuno ODIM HDF5 radar volume scans into the observation format required by WRFDA Radar Data Assimilation.
https://docs.wradlib.org/projects/notebooks/en/latest/notebooks/fileio/backends/odim_backend.html

Objectives

The objective of this module is to

- read Furuno ODIM HDF5 radar volume scans,
- extract radar reflectivity (DBZH) and radial velocity (VRAD),
- interpolate radar observations onto a regular Cartesian grid,
- convert the gridded observations into the WRFDA FM-128 Radar Observation format,
- automatically generate domain-specific observation files,
- optionally upload the observation files to an FTP server for operational forecasting.

## Step 4a. Convert Radar PPI to NetCDF

Key processing steps include:

```python
    # SKIP JIKA SUDAH PERNAH DICONVERT

        vol.latitude.values

        ygrid

        xx /
        (
            R *
            np.cos(
                np.deg2rad(radar_lat)
            )
        )

    latGrid = radar_lat + (
        yy / R

    lats = latGrid[:,0].astype(np.float32)

    nlat = len(lats)
    nlon = len(lons)
    nlev = len(levs)

        (nlat,nlon,nlev),
        np.nan,

        # -----------------------------------

        # -----------------------------------
        # GRID INDEX
        # -----------------------------------

        ix = np.round(
            (x - xgrid[0]) /
            grid_res
        ).astype(int)

        iy = np.round(
            (y - ygrid[0]) /
            grid_res
        ).astype(int)

        valid = (
            (ix >= 0)
            &
            (ix < len(xgrid))
            &
            (iy >= 0)
            &
            (iy < len(ygrid))
        )

        ix = ix[valid]
        iy = iy[valid]

        dbzv = dbz[valid]
        vradv = vrad[valid]

        grd_dbz = np.full(
            xx.shape,
            np.nan,
            dtype=np.float32
        )

        grd_v = np.full(
            xx.shape,
            np.nan,
            dtype=np.float32
        )

        for k in range(len(ix)):

            # DBZH

            if np.isfinite(dbzv[k]):

                if np.isnan(
                    grd_dbz[iy[k],ix[k]]
                ):

                    grd_dbz[
                        iy[k],
                        ix[k]
                    ] = dbzv[k]

                else:

                    grd_dbz[
                        iy[k],
                        ix[k]
                    ] = max(
                        grd_dbz[
                            iy[k],
                            ix[k]
                        ],
                        dbzv[k]
                    )

            # VRAD

            grd_v[
                iy[k],
                ix[k]
            ] = vradv[k]

        dbz3d[:,:,sw] = grd_dbz
        vrad3d[:,:,sw] = grd_v

    # =====================================================
    # OUTPUT
    # =====================================================

    out_dbz = os.path.join(
        outdir,
        f"{timestep}_dbz.nc"
    )
    
    out_v = os.path.join(
        outdir,
        f"{timestep}_v.nc"
    )

    write_nc(
        out_dbz,
        "dBZ",
        dbz3d,
        lons,
        lats,
        levs,
        fname
    )

    write_nc(
        out_v,
        "VRAD",
        vrad3d,
        lons,
        lats,
        levs,
        fname
    )

    print("DBZ :", out_dbz)
    print("VRAD:", out_v)
    print(
        "Elapsed:",
        dt.datetime.now() - t0
    )

print("")
print("ALL FINISHED")
```
